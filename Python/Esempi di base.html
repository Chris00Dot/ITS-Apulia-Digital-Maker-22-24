class Cane:
    def __init__(self, nome, eta, tipologia):
        self.nome = nome
        self.eta = eta
        self.tipologia = tipologia
    def parla(self):
        print('Bau dice ' + self.nome)
    def __del__(self):
        print('Bye '+self.nome)
